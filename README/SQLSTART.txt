1. Tabellen erstellen im Query Tool ( auch in der Reihenfolge )
    create table users( id Serial Primary Key, email varchar(255), lastname varchar(255), passwort varchar(255), birthdate date, firstname varchar(255) )
    create table tierheim (thid serial Primary Key, thname varchar(255), thadresse varchar(255) )
    Create Table tiere (tid serial Primary Key, tname varchar(255), tierart varchar(255), rasse varchar(255), talter int , thid int , Foreign Key (thid) references tierheim(thid) )
